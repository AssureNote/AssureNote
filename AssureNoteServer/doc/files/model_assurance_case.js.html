/**
* @method getMd5String
* @return {String}
*/
function getMd5String() {
    var d = new Date();
    var md5 = crypto.createHash('md5');

    md5.update(d.toString() + d.getMilliseconds(), 'utf-8');
    return md5.digest('hex');
}
exports.getMd5String = getMd5String;
